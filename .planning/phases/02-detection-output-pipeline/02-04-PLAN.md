    build_region_table(df, region):
      if df.empty:
          return ""
      top = df.head(10).reset_index(drop=True)
      rows = []
      if region == "EU":
          header = "| Rank | SKU | Country | Velocity/wk | +$/unit | Impact/4wk | Days since case |"
          separator = "|------|-----|---------|------------|---------|-----------|----------------|"
          for i, row in top.iterrows():
              days = f"{int(row['days_since_last_case'])}d" if pd.notna(row.get('days_since_last_case')) else "N/A"
              rows.append(f"| {i+1} | {row['sku']} | {row['country']} | {row['velocity_w3']:.0f} | ${row['overcharge_per_unit']:.2f} | ${row['impact_4w']:.0f} | {days} |")
      else:
          header = "| Rank | SKU | Velocity/wk | +$/unit | Impact/4wk | Days since case |"
          separator = "|------|-----|------------|---------|-----------|----------------|"
          for i, row in top.iterrows():

              rows.append(f"| {i+1} | {row['sku']} | {row['velocity_w3']:.0f} | ${row['overcharge_per_unit']:.2f} | ${row['impact_4w']:.0f} | {days} |")
      return "\n".join([header, separator] + rows)

    build_anomaly_json(region_df, region, run_date, slots_remaining):
      top_skus = []
      for _, row in region_df.head(10).iterrows():
          top_skus.append({
              "sku": row.get("sku", row.get("key_sales_marketplace_sku", "")),
              "country": row.get("country", ""),
              "overcharge": round(float(row["overcharge_per_unit"]), 2),
              "impact_4w": round(float(row["impact_4w"]), 0),
              "velocity_w3": int(row.get("velocity_w3", 0)),
              "days_since_case": int(row["days_since_last_case"]) if pd.notna(row.get("days_since_last_case")) else None,
          })
      return {
          "region": region,
          "run_date": str(run_date),
          "slots_used": 10 - slots_remaining,
          "top_skus": top_skus,
      }

    generate_narrative(region_summaries):
      client = anthropic.Anthropic()  # reads ANTHROPIC_API_KEY from env per PATTERNS.md
      SYSTEM_PROMPT = (
          "You are summarizing an Amazon FBA fee audit run for the OrganiHaus data team. "
          "For each region, generate ONE paragraph of exactly <=50 words. "
          "Cover the most impactful overcharge SKUs by dollar impact. "
          "Use 'may indicate' not 'is caused by'. Plain prose, no bullet points, no headings. "
          "Return ONLY the paragraph text for each region, nothing else."
      )
      narratives = {}
      for region, summary in region_summaries.items():
          user_msg = f"Region {region} anomaly summary:\n{json.dumps(summary, indent=2)}"
          response = client.messages.create(
              model="claude-sonnet-4-6",
              max_tokens=200,
              system=SYSTEM_PROMPT,
              messages=[{"role": "user", "content": user_msg}],
          )
          narratives[region] = response.content[0].text
      return narratives
    Note per D-20: single call per region, no tools, no streaming, context well under 4K tokens.

    build_comment_body(regional_tables, narratives):
      ESCALATION_PROMPT = "Reply YES to this comment to open an investigation task for the top flagged SKUs."
      sections = []
      for region in ["US", "CA", "GB", "EU"]:
          table = regional_tables.get(region, "")
          if not table:

          sections.append(f"## {region}\n{narrative}\n\n{table}")
      body = "\n\n".join(sections)
      return f"{body}\n\n{ESCALATION_PROMPT}"

    post_clickup_comment(task_id, comment_text, api_key):
      url = f"https://api.clickup.com/api/v2/task/{task_id}/comment"
      headers = {
          "Authorization": api_key,        # raw pk_ token, NO "Bearer" prefix (PITFALLS §1)
          "Content-Type": "application/json",
      }
      body = {"comment_text": comment_text, "notify_all": False}
      r = requests.post(url, headers=headers, json=body, timeout=30)
      r.raise_for_status()  # raises on 4xx/5xx (PATTERNS.md HTTP error handling)
      print(f"ClickUp comment posted to task {task_id}")
      return r.json()

    attach_csv_to_task(task_id, csv_bytes, run_date, api_key):
      url = f"https://api.clickup.com/api/v2/task/{task_id}/attachment"
      headers = {"Authorization": api_key}  # NO Content-Type — requests sets multipart boundary (PITFALLS §2)
      filename = f"anomaly_report_{run_date.strftime('%Y%m%d')}.csv"
      files = {"attachment": (filename, csv_bytes, "text/csv")}
      r = requests.post(url, headers=headers, files=files, timeout=60)
      r.raise_for_status()
      print(f"CSV attached to ClickUp task {task_id}: {filename}")
      return r.json()
