    load_cases_csv():
      if not CASES_CSV_PATH.exists():
          print(f"WARNING: Cases CSV not found at {CASES_CSV_PATH} — slot/eligibility checks will return defaults")
          return pd.DataFrame(columns=['Region', 'SKU', 'Date Closed', 'Date Created'])
      df = pd.read_csv(

          parse_dates=['Date Created', 'Date Closed'],
          dayfirst=True,
      )
      return df

    get_slots_remaining(cases_df, region, monthly_limit, run_date):
      if cases_df.empty:
          return monthly_limit
      month_start = pd.Timestamp(run_date.replace(day=1))
      run_ts = pd.Timestamp(run_date)
      mask = (
          (cases_df['Region'] == region) &
          (cases_df['Date Closed'] >= month_start) &
          (cases_df['Date Closed'] <= run_ts)
      )
      used = mask.sum()
      return max(0, monthly_limit - int(used))

    get_sku_eligible(cases_df, sku, cap_days, cap_count, run_date):
      if cases_df.empty:
          return True
      cutoff = pd.Timestamp(run_date) - pd.Timedelta(days=cap_days)
      run_ts = pd.Timestamp(run_date)
      mask = (
          (cases_df['SKU'] == sku) &
          (cases_df['Date Closed'] >= cutoff) &
          (cases_df['Date Closed'] <= run_ts)
      )
      return int(mask.sum()) < cap_count

    add_eligibility(scored_df, cases_df, config, run_date):
      df = scored_df.copy()
      fnsku_cap_days = config['FNSKU_CAP_DAYS']
      fnsku_cap_count = 2  # hardcoded per D-30c: "< 2 cases closed per SKU in last 60 days"
      monthly_limit = config['MONTHLY_CASE_LIMIT']

      # days_since_last_case: days since the most recent Date Closed for this SKU
      run_ts = pd.Timestamp(run_date)
      def _days_since(sku):
          if cases_df.empty:
              return None
          sku_cases = cases_df[cases_df['SKU'] == sku]['Date Closed'].dropna()
          if sku_cases.empty:
              return None
          return int((run_ts - sku_cases.max()).days)
      df['days_since_last_case'] = df['sku'].map(_days_since)

      # eligible_for_case: sku_eligible AND slots_remaining[region] > 0
      slots_cache = {}
      def _eligible(row):
          region = row['sales_region']
          if region not in slots_cache:
              slots_cache[region] = get_slots_remaining(cases_df, region, monthly_limit, run_date)
          sku_ok = get_sku_eligible(cases_df, row['sku'], fnsku_cap_days, fnsku_cap_count, run_date)
          return sku_ok and slots_cache[region] > 0
      df['eligible_for_case'] = df.apply(_eligible, axis=1)
      return df
